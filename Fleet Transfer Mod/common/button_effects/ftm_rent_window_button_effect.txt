# условие отображения кнопки "купить"
ftm_rent_slot_1_buy = {
	potential = {
		has_global_flag = ftm_rent_slot_1_selected
		NOT = {
			is_same_value = event_target:ftm_rented_fleet_slot_1_seller
		}
		has_communications = event_target:ftm_rented_fleet_slot_1_seller
	}
	allow = {
		always = yes
	}
	effect = {
		custom_tooltip = ftm_rent_slot_1_buy_info
		hidden_effect = {
			log = "[Root.GetName] want to buy fleet in slot 1."

		### energy check
			event_target:ftm_rented_fleet_slot_1_seller = { 
				set_variable = {
					which = ftm_buyer_rented_fleet_slot_1_energy_current_count
					value = 0
				}
			}
			set_variable = {
				which = ftm_buyer_rented_fleet_slot_1_check_energy
				value = 0
			}
			save_event_target_as = ftm_buyer_rented_fleet_slot_1_checker
			export_resource_stockpile_to_variable = { resource = energy variable = ftm_buyer_rented_fleet_slot_1_check_energy }
			event_target:ftm_rented_fleet_slot_1_seller = {
				set_variable = {
					which = ftm_buyer_rented_fleet_slot_1_energy_current_count
					value = event_target:ftm_buyer_rented_fleet_slot_1_checker.ftm_buyer_rented_fleet_slot_1_check_energy
				}
			}
			log = "[Root.GetName]: energy_check complete. ENERGY: [event_target:ftm_rented_fleet_slot_1_seller.ftm_buyer_rented_fleet_slot_1_energy_current_count]."
			
			
			### 
			
			#4
			if = {
				limit = {
					has_country_flag = ftm_rent_slot_1_sanction_block
				}
				log = "Check [Root.GetName] sanction: SANCTION ACTIVE!!!"
				country_event = { id = ftm_rent.121 }
			}
			if = {
				limit = {
					NOT = {
						has_country_flag = ftm_rent_slot_1_sanction_block
					}
					event_target:ftm_rented_fleet_slot_1_seller = { 
						check_variable = {
							which = ftm_buyer_rented_fleet_slot_1_energy_current_count
							value < ftm_rented_fleet_slot_1_cost
						}
					}
				}
				log = "Check [Root.GetName] energy: need [event_target:ftm_rented_fleet_slot_1_seller.ftm_rented_fleet_slot_1_cost]... FAIL. NO ENERGY."
				country_event = { id = ftm_rent.105 }
			}
			if = {
				limit = {
					NOT = {
						has_country_flag = ftm_rent_slot_1_sanction_block
					}
					event_target:ftm_rented_fleet_slot_1_seller = { 
						check_variable = {
							which = ftm_buyer_rented_fleet_slot_1_energy_current_count
							value >= ftm_rented_fleet_slot_1_cost
						}
					}
				}
				log = "Check [Root.GetName] energy: need [event_target:ftm_rented_fleet_slot_1_seller.ftm_rented_fleet_slot_1_cost]... SUCCESS. ENERGY IN."
				
				### ПРОВЕРКА ОТНОШЕНИЙ С ИИ-ПРОДАВЦОМ
				log = "Checking [Root.GetName] and [event_target:ftm_rented_fleet_slot_1_seller.GetName] opinion..."
				
				# продавец - ИИ, отношение выше -50
				if = {
					limit = {
						AND = {
							NOT = {
								event_target:ftm_rented_fleet_slot_1_seller = {
									OR = {
										is_rival = prev
										opinion = {
											who = prev
											value < -50
										}
										their_opinion = {
											who = prev
											value < -50
										}
									}
								}
							}
							event_target:ftm_rented_fleet_slot_1_seller = {
								is_ai = yes
							}
						}
					}
					log = "[event_target:ftm_rented_fleet_slot_1_seller.GetName] is AI. Opinion GOOD."
					save_global_event_target_as = ftm_rent_slot_1_buyer
					
					# передача флота и денег
					event_target:ftm_rented_fleet_slot_1 = { 
						set_fleet_flag = ftm_rented_fleet_slot_1
						set_owner = event_target:ftm_rent_slot_1_buyer
						set_event_locked = no
					}
					event_target:ftm_rent_slot_1_buyer = { 
						country_event = { id = ftm_rent.111 }
					
					
						# снятие ресурсов при покупке			
				
						event_target:ftm_rented_fleet_slot_1_seller = { 
							set_variable = {
								which = ftm_rented_fleet_slot_1_lot_cost
								value = ftm_rented_fleet_slot_1_cost
							}
						}
						log = "[Root.GetName]: [event_target:ftm_rented_fleet_slot_1_seller.ftm_rented_fleet_slot_1_lot_cost] energy is SLOT-1 cost"	
						while = {
							count = 99
							if = {
								limit = {
									event_target:ftm_rented_fleet_slot_1_seller = { 
										check_variable = {
											which = ftm_rented_fleet_slot_1_lot_cost
											value >= 10000
										}
									}
								}
								add_resource = { energy = -10000 }
								event_target:ftm_rented_fleet_slot_1_seller = { 
									change_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value = -10000
									}
								}
							}
							else_if = {
								limit = {
									event_target:ftm_rented_fleet_slot_1_seller = { 
										check_variable = {
											which = ftm_rented_fleet_slot_1_lot_cost
											value >= 1000
										}
									}
								}
								add_resource = { energy = -1000 }
								event_target:ftm_rented_fleet_slot_1_seller = { 
									change_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value = -1000
									}		
								}
							}
							else_if = {
								limit = {
									event_target:ftm_rented_fleet_slot_1_seller = { 
										check_variable = {
											which = ftm_rented_fleet_slot_1_lot_cost
											value >= 100
										}
									}
								}
								add_resource = { energy = -100 }
								event_target:ftm_rented_fleet_slot_1_seller = { 
									change_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value = -100
									}
								}					
							}
							else_if = {
								limit = {
									event_target:ftm_rented_fleet_slot_1_seller = { 
										check_variable = {
											which = ftm_rented_fleet_slot_1_lot_cost
											value >= 10
										}
									}
								}
								add_resource = { energy = -10 }
								event_target:ftm_rented_fleet_slot_1_seller = { 
									change_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value = -10
									}
								}
							}
							else_if = {
								limit = {
									event_target:ftm_rented_fleet_slot_1_seller = { 
										check_variable = {
											which = ftm_rented_fleet_slot_1_lot_cost
											value >= 1
										}
									}
								}
								add_resource = { energy = -1 }
								event_target:ftm_rented_fleet_slot_1_seller = { 
									change_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value = -1
									}					
								}
							}
							else = {
								#nothing
							}
						}	
					}
					event_target:ftm_rented_fleet_slot_1_seller = { 
						country_event = { id = ftm_rent.112 }
					}
					remove_global_flag = ftm_rent_slot_1_selected
					set_global_flag = ftm_rent_slot_1_arended	
				}
				
				# продавец - ИИ, отношение плохое
				if = {
					limit = {
						AND = {
							event_target:ftm_rented_fleet_slot_1_seller = {
								OR = {
									is_rival = prev
									opinion = {
										who = prev
										value < -50
									}
									their_opinion = {
										who = prev
										value < -50
									}
								}
							}
							event_target:ftm_rented_fleet_slot_1_seller = {
								is_ai = yes
							}
						}
					}
					log = "[event_target:ftm_rented_fleet_slot_1_seller.GetName] is AI. Opinion BAD."
					country_event = { id = ftm_rent.125 }
				}
				
				# продавец - игрок
				if = {
					limit = {
						event_target:ftm_rented_fleet_slot_1_seller = {
							is_ai = no
						}
					}
					log = "[event_target:ftm_rented_fleet_slot_1_seller.GetName] is PLAYER. Sending request..."
					save_event_target_as = ftm_rent_slot_1_requester
					
				
					# снятие ресурсов 	
			
					event_target:ftm_rented_fleet_slot_1_seller = { 
						set_variable = {
							which = ftm_rented_fleet_slot_1_lot_cost
							value = ftm_rented_fleet_slot_1_cost
						}
					}	
					while = {
						count = 99
						if = {
							limit = {
								event_target:ftm_rented_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value >= 10000
									}
								}
							}
							add_resource = { energy = -10000 }
							event_target:ftm_rented_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_rented_fleet_slot_1_lot_cost
									value = -10000
								}
							}
						}
						else_if = {
							limit = {
								event_target:ftm_rented_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value >= 1000
									}
								}
							}
							add_resource = { energy = -1000 }
							event_target:ftm_rented_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_rented_fleet_slot_1_lot_cost
									value = -1000
								}		
							}
						}
						else_if = {
							limit = {
								event_target:ftm_rented_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value >= 100
									}
								}
							}
							add_resource = { energy = -100 }
							event_target:ftm_rented_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_rented_fleet_slot_1_lot_cost
									value = -100
								}
							}					
						}
						else_if = {
							limit = {
								event_target:ftm_rented_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value >= 10
									}
								}
							}
							add_resource = { energy = -10 }
							event_target:ftm_rented_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_rented_fleet_slot_1_lot_cost
									value = -10
								}
							}
						}
						else_if = {
							limit = {
								event_target:ftm_rented_fleet_slot_1_seller = { 
									check_variable = {
										which = ftm_rented_fleet_slot_1_lot_cost
										value >= 1
									}
								}
							}
							add_resource = { energy = -1 }
							event_target:ftm_rented_fleet_slot_1_seller = { 
								change_variable = {
									which = ftm_rented_fleet_slot_1_lot_cost
									value = -1
								}					
							}
						}
						else = {
							#nothing
						}
					}
					
					country_event = { id = ftm_rent.126 }
					event_target:ftm_rented_fleet_slot_1_seller = {
						country_event = { id = ftm_rent.127 }
					}
				}
			}
		}
	}
}
