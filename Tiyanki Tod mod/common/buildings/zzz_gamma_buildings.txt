		}

		category = planet_buildings

		# Base cost
		cost = {
			minerals = 500
		}
		# Org/Lith cost
		cost = {
			trigger = {
				exists = owner
				owner = { is_lithoid_empire = no }
			}
			food = 500
		}
		cost = {
			trigger = {
				exists = owner
				owner = { is_lithoid_empire = yes }
			}
			energy = 500
		}

		upkeep = {
			energy = 2
		}
		# Org/Lith upkeep

			trigger = {
				exists = owner

		}

			trigger = {
				exists = owner
				owner = { is_lithoid_empire = yes }

		}
